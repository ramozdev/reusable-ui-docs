# Badge

> Badges are subtle elements to highlight information and metrics.

import * as Badge from "@/ui/html/badge";
import { colors } from "@/lib/colors";

export function Default() {
  return (

        <Badge.Solid key={`badge-solid-default-${color}`} color={color}>

## Usage

### Solid

| Prop        | `color`                                                                                                                                                                                                                                                                          |
| ----------- | -------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------- |
| Description | The badge's color.                                                                                                                                                                                                                                                               |
| type        | `'brand' \| 'slate' \| 'gray' \| 'zinc' \| 'neutral' \| 'stone' \| 'red' \| 'orange' \| 'amber' \| 'yellow' \| 'lime' \| 'green' \| 'emerald' \| 'teal' \| 'cyan' \| 'sky' \| 'blue' \| 'indigo' \| 'violet' \| 'purple' \| 'fuchsia' \| 'pink' \| 'rose' \| 'white-black'{:ts}` |
| default     | `'blue'{:ts}`                                                                                                                                                                                                                                                                    |
| required    | `false{:ts}`                                                                                                                                                                                                                                                                     |

| Prop        | `variant`                                |
| ----------- | ---------------------------------------- |
| Description | The badge's style.                       |
| type        | `'default' \| 'outline' \| 'ghost'{:ts}` |
| default     | `'default'{:ts}`                         |
| required    | `false{:ts}`                             |
