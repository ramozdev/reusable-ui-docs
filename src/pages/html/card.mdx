import { Card } from "@/ui/html/card";
import { colors } from "@/lib/colors";

export function Outline() {
  return (
    <div className="grid gap-4 sm:grid-cols-2 lg:grid-cols-4">
      {colors.map((color, index) => (
        <Card key={`card-outline-${color}-${index}`} color={color}>

```tsx copy
import { Card } from "@/ui/html/card";
import { colors } from "@/lib/colors";

export function Plane() {
  return (
    <div className="grid gap-4 sm:grid-cols-2 lg:grid-cols-4">
      {colors.map((color, index) => (
        <Card
          key={`card-outline-${color}-${index}`}
          variant="plane"
          color={color}
        >
          {color}
        </Card>
      ))}
    </div>
  );
}
```

## Usage

### Card

| Prop        | `color`                                                                                                                                                                                                                                                                          |
| ----------- | -------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------- |
| Description | The card's color.                                                                                                                                                                                                                                                                |
| type        | `'brand' \| 'slate' \| 'gray' \| 'zinc' \| 'neutral' \| 'stone' \| 'red' \| 'orange' \| 'amber' \| 'yellow' \| 'lime' \| 'green' \| 'emerald' \| 'teal' \| 'cyan' \| 'sky' \| 'blue' \| 'indigo' \| 'violet' \| 'purple' \| 'fuchsia' \| 'pink' \| 'rose' \| 'white-black'{:ts}` |
| default     | `'white-black'{:ts}`                                                                                                                                                                                                                                                             |
| required    | `false{:ts}`                                                                                                                                                                                                                                                                     |

| Prop        | `variant`                   |
| ----------- | --------------------------- |
| Description | The card's style.           |
| type        | `'outline' \| 'plane'{:ts}` |
| default     | `'outline'{:ts}`            |
| required    | `false{:ts}`                |
