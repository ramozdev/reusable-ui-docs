
## Demo

```tsx copy filename="@/demos/radix/checkbox.tsx"
import * as Checkbox from "@/ui/radix/checkbox";
import { CheckIcon } from "@radix-ui/react-icons";

export function CheckboxDemo() {
  return (
    <form>
      <div className="flex items-center">
        <Checkbox.Root defaultChecked id="c1">
          <Checkbox.Indicator>
            <CheckIcon />
          </Checkbox.Indicator>
        </Checkbox.Root>
        <label className="pl-[15px] text-[15px] leading-none" htmlFor="c1">
          Accept terms and conditions.
        </label>
      </div>
    </form>
  );
}
```

## Usage

Reusable UI extends the API of Radix UI to provide a more convenient API. This seciton contains the additions made to the API.

### Indicator

| Prop        | `color`                                                                                                                                                                                                                                                                          |
| ----------- | -------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------- |
| Description | The indicator's color.                                                                                                                                                                                                                                                           |
| type        | `'brand' \| 'slate' \| 'gray' \| 'zinc' \| 'neutral' \| 'stone' \| 'red' \| 'orange' \| 'amber' \| 'yellow' \| 'lime' \| 'green' \| 'emerald' \| 'teal' \| 'cyan' \| 'sky' \| 'blue' \| 'indigo' \| 'violet' \| 'purple' \| 'fuchsia' \| 'pink' \| 'rose' \| 'white-black'{:ts}` |
| default     | `'neutral'{:ts}`                                                                                                                                                                                                                                                                 |
| required    | `false{:ts}`                                                                                                                                                                                                                                                                     |

### Root

| Prop        | `color`                                                                                                                                                                                                                                                                          |
| ----------- | -------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------- |
| Description | The root's color.                                                                                                                                                                                                                                                                |
| type        | `'brand' \| 'slate' \| 'gray' \| 'zinc' \| 'neutral' \| 'stone' \| 'red' \| 'orange' \| 'amber' \| 'yellow' \| 'lime' \| 'green' \| 'emerald' \| 'teal' \| 'cyan' \| 'sky' \| 'blue' \| 'indigo' \| 'violet' \| 'purple' \| 'fuchsia' \| 'pink' \| 'rose' \| 'white-black'{:ts}` |
| default     | `'blue'{:ts}`                                                                                                                                                                                                                                                                    |
| required    | `false{:ts}`                                                                                                                                                                                                                                                                     |
