
## Demo

```tsx copy filename="@/demos/radix/tooltip.tsx"
import * as Button from "@/ui/html/button";
import * as Tooltip from "@/ui/radix/tooltip";
import { PlusIcon } from "@radix-ui/react-icons";

export function TooltipDemo() {
  return (
    <Tooltip.Provider>
      <Tooltip.Root>
        <Tooltip.Trigger asChild>
          <Button.Solid className="rounded-full">
            <PlusIcon />
          </Button.Solid>
        </Tooltip.Trigger>
        <Tooltip.Portal>
          <Tooltip.Content>
            Add to library
            <Tooltip.Arrow />
          </Tooltip.Content>
        </Tooltip.Portal>
      </Tooltip.Root>
    </Tooltip.Provider>
  );
}
```

## Usage

Reusable UI extends the API of Radix UI to provide a more convenient API. This seciton contains the additions made to the API.

### Arrow

| ----------- | -------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------- |
| Description | The arrow's color.                                                                                                                                                                                                                                                               |

| default     | `'neutral'{:ts}`                                                                                                                                                                                                                                                                 |
| required    | `false{:ts}`                                                                                                                                                                                                                                                                     |
