## Demo

```tsx copy filename="@/demos/radix/dialog.tsx"
import * as Dialog from "@/ui/radix/dialog";
import { Cross2Icon } from "@radix-ui/react-icons";
import * as Button from "@/ui/html/button";
import { Input } from "@/ui/html/input";

export function DialogDemo() {
  return (
    <Dialog.Root>
      <Dialog.Trigger asChild>
        <Button.Solid color="blue">Edit profile</Button.Solid>
      </Dialog.Trigger>
      <Dialog.Portal>
        <Dialog.Overlay />
        <Dialog.Content>
          <Dialog.Title>Edit profile</Dialog.Title>
          <Dialog.Description>
            Make changes to your profile here. Click save when you&apos;re done.
          </Dialog.Description>
          <fieldset className="mb-[15px] flex items-center gap-5">
            <label htmlFor="name">Name</label>
            <Input id="name" defaultValue="Diego Ramos" />
          </fieldset>
          <fieldset className="mb-[15px] flex items-center gap-5">
            <label htmlFor="username">Username</label>
            <Input id="username" defaultValue="@ramozdev" />
          </fieldset>
          <div className="mt-[25px] flex justify-end">
            <Dialog.Close asChild>
              <Button.Subtle color="green">Save changes</Button.Subtle>
            </Dialog.Close>
          </div>
          <Dialog.Close asChild>
            <Button.Subtle
              color="neutral"
              className="absolute right-[10px] top-[10px] rounded-full"
              aria-label="Close"
            >
              <Cross2Icon />
            </Button.Subtle>
          </Dialog.Close>
        </Dialog.Content>
      </Dialog.Portal>
    </Dialog.Root>
  );
}
```

## Usage

Reusable UI extends the API of Radix UI to provide a more convenient API. This seciton contains the additions made to the API.

### Content

| Prop        | `color`                                                                                                                                                                                                                                                                          |
| ----------- | -------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------- |
| Description | The content's color.                                                                                                                                                                                                                                                             |
| type        | `'brand' \| 'slate' \| 'gray' \| 'zinc' \| 'neutral' \| 'stone' \| 'red' \| 'orange' \| 'amber' \| 'yellow' \| 'lime' \| 'green' \| 'emerald' \| 'teal' \| 'cyan' \| 'sky' \| 'blue' \| 'indigo' \| 'violet' \| 'purple' \| 'fuchsia' \| 'pink' \| 'rose' \| 'white-black'{:ts}` |
| default     | `'white-black'{:ts}`                                                                                                                                                                                                                                                             |
| required    | `false{:ts}`                                                                                                                                                                                                                                                                     |
