## Demo

```tsx copy filename="@/demos/radix/collapsible.tsx"
import React from "react";
import * as Collapsible from "@/ui/radix/collapsible";
import { RowSpacingIcon, Cross2Icon } from "@radix-ui/react-icons";
import * as Button from "@/ui/html/button";

export function CollapsibleDemo() {
  const [open, setOpen] = React.useState(false);
  return (
    <Collapsible.Root open={open} onOpenChange={setOpen}>
      <div
        style={{
          display: "flex",
          alignItems: "center",
          justifyContent: "space-between",
        }}
      >
        <span className="text-[15px] leading-[25px]">
          @peduarte starred 3 repositories
        </span>
        <Collapsible.Trigger asChild>
          <Button.Solid
            color="neutral"
            variant="outline"
            className="inline-flex h-[25px] w-[25px] items-center justify-center rounded-full p-0"
          >
            {open ? <Cross2Icon /> : <RowSpacingIcon />}
          </Button.Solid>
        </Collapsible.Trigger>
      </div>

      <div className="my-[10px] rounded-md bg-white p-[10px] dark:bg-black">
        <span className="text-[15px] leading-[25px]">@radix-ui/primitives</span>
      </div>

      <Collapsible.Content>
        <div className="my-[10px] rounded-md bg-white p-[10px] dark:bg-black">
          <span className="text-[15px] leading-[25px]">@radix-ui/colors</span>
        </div>
        <div className="my-[10px] rounded-md bg-white p-[10px] dark:bg-black">
          <span className="text-[15px] leading-[25px]">@stitches/react</span>
        </div>
      </Collapsible.Content>
    </Collapsible.Root>
  );
}
```
